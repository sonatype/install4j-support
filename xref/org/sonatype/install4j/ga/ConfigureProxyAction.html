13  package org.sonatype.install4j.ga;
14  
15  import com.dmurph.tracking.JGoogleAnalyticsTracker;
16  import com.install4j.api.context.Context;
17  import com.install4j.runtime.beans.actions.update.DownloadFileAction;
18  
19  import java.io.File;
20  
21  /**
22   * Configure proxy for Google Analytics support.
23   *
24   * @since 1.0
25   */
26  public class ConfigureProxyAction
27      extends ActionSupport
28  {
29      @Override
30      protected boolean execute(final Context context) throws Exception {
31          if (GoogleAnalyticsActionSupport.isDisabled()) {
32              log.debug("Google Analytics disabled; skipping proxy detection");
33              return true;
34          }
35  
36          // Trigger a file download action, so that proxy configuration can be detected
37          log.debug("Detecting proxy configuration");
38          DownloadFileAction action = new DownloadFileAction();
39          action.setUrl("http://www.google-analytics.com/__utm.gif"); // just use the tracker's base url w/o any params
40          action.setAskForProxy(true);
41          action.setDeleteOnExit(true);
42          action.setRetryIfInterrupted(true);
43          action.setShowError(false);
44          action.setTargetFile(File.createTempFile("configure-proxy", ".tmp").getAbsolutePath());
45          action.execute(context);
46  
47          if (log.isDebugEnabled()) {
48              log.debug("proxySet: {}", System.getProperty("proxySet"));
49              log.debug("proxyHost: {}", System.getProperty("proxyHost"));
50              log.debug("proxyPort: {}", System.getProperty("proxyPort"));
51          }
52  
53          boolean proxySet = Boolean.getBoolean("proxySet");
54          if (proxySet) {
55              String proxyHost = System.getProperty("proxyHost");
56              String proxyPort = System.getProperty("proxyPort");
57              if (proxyHost == null && proxyPort == null) {
58                  log.warn("Invalid proxy configuration; proxySet=true, but proxyHost and proxyPort missing");
59                  return true;
60              }
61  
62              String proxy = String.format("%s:%s", proxyHost, proxyPort);
63              log.debug("Proxy: {}", proxy);
64              JGoogleAnalyticsTracker.setProxy(proxy);
65          }
66  
67          return true;
68      }
69  }
