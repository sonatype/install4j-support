13  package org.sonatype.install4j.ga;
14  
15  import com.install4j.api.beaninfo.BeanValidationException;
16  import com.install4j.api.beans.Bean;
17  
18  import static com.install4j.api.beaninfo.Install4JPropertyDescriptor.create;
19  
20  /**
21   * Action bean-info for {@link TrackEventAction}.
22   *
23   * @since 1.0
24   */
25  public class TrackEventActionBeanInfo
26      extends ActionBeanInfoSupport
27  {
28    private static final String PROPERTY_TRACKING_CODE = "trackingCode";
29  
30    private static final String PROPERTY_CATEGORY = "category";
31  
32    private static final String PROPERTY_ACTION = "action";
33  
34    private static final String PROPERTY_LABEL = "label";
35  
36    private static final String PROPERTY_VALUE = "value";
37  
38    public TrackEventActionBeanInfo() {
39      super(
40          "Track Event",
41          "Send a Google Analytics tracking event",
42          CATEGORY_GA,
43          MULTIPLE_INSTANCES,
44          INSTALLED_FILES_NOT_REQUIRED,
45          DEFAULT_SORT_KEY,
46          TrackEventAction.class
47      );
48  
49      addPropertyDescriptor(create(
50          PROPERTY_TRACKING_CODE,
51          getBeanClass(),
52          "Tracking Code",
53          "Tracking code"
54      ));
55  
56      addPropertyDescriptor(create(
57          PROPERTY_CATEGORY,
58          getBeanClass(),
59          "Category",
60          "Event category"
61      ));
62  
63      addPropertyDescriptor(create(
64          PROPERTY_ACTION,
65          getBeanClass(),
66          "Action",
67          "Event action"
68      ));
69  
70      addPropertyDescriptor(create(
71          PROPERTY_LABEL,
72          getBeanClass(),
73          "Label",
74          "Event label"
75      ));
76  
77      addPropertyDescriptor(create(
78          PROPERTY_VALUE,
79          getBeanClass(),
80          "Value",
81          "Event value"
82      ));
83    }
84  
85    @Override
86    public void validateBean(final Bean bean) throws BeanValidationException {
87      checkNotEmpty(PROPERTY_TRACKING_CODE, bean);
88      checkNotEmpty(PROPERTY_CATEGORY, bean);
89      checkNotEmpty(PROPERTY_ACTION, bean);
90      // label is optional
91      // value is optional
92    }
93  }
