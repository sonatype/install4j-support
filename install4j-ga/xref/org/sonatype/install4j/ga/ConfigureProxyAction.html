13  package org.sonatype.install4j.ga;
14  
15  import java.io.File;
16  
17  import com.dmurph.tracking.JGoogleAnalyticsTracker;
18  import com.install4j.api.context.Context;
19  import com.install4j.runtime.beans.actions.net.DownloadFileAction;
20  
21  /**
22   * Configure proxy for Google Analytics support.
23   *
24   * @since 1.0
25   */
26  public class ConfigureProxyAction
27      extends ActionSupport
28  {
29    @Override
30    protected boolean execute(final Context context) throws Exception {
31      if (GoogleAnalyticsActionSupport.isDisabled()) {
32        log.debug("Google Analytics disabled; skipping proxy detection");
33        return true;
34      }
35  
36      // Trigger a file download action, so that proxy configuration can be detected
37      log.debug("Detecting proxy configuration");
38      DownloadFileAction action = new DownloadFileAction();
39      action.setUrl("http://www.google-analytics.com/__utm.gif"); // just use the tracker's base url w/o any params
40      action.setAskForProxy(true);
41      action.setDeleteOnExit(true);
42      action.setRetryIfInterrupted(true);
43      action.setShowFileName(false);
44      action.setShowProgress(false);
45      action.setShowError(false);
46      action.setTargetFile(File.createTempFile("configure-proxy", ".tmp"));
47      action.execute(context);
48  
49      if (log.isDebugEnabled()) {
50        log.debug("proxySet: {}", System.getProperty("proxySet"));
51        log.debug("proxyHost: {}", System.getProperty("proxyHost"));
52        log.debug("proxyPort: {}", System.getProperty("proxyPort"));
53      }
54  
55      boolean proxySet = Boolean.getBoolean("proxySet");
56      if (proxySet) {
57        String proxyHost = System.getProperty("proxyHost");
58        String proxyPort = System.getProperty("proxyPort");
59        if (proxyHost == null && proxyPort == null) {
60          log.warn("Invalid proxy configuration; proxySet=true, but proxyHost and proxyPort missing");
61          return true;
62        }
63  
64        String proxy = String.format("%s:%s", proxyHost, proxyPort);
65        log.debug("Proxy: {}", proxy);
66        JGoogleAnalyticsTracker.setProxy(proxy);
67      }
68  
69      return true;
70    }
71  }
