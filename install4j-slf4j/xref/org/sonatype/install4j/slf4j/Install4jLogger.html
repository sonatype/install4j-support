65    private static String getStringProperty(final String name) {
66      String prop = null;
67      try {
68        prop = System.getProperty(name);
69      }
70      catch (SecurityException e) {
71        // Ignore
72      }
73      return (prop == null) ? configuration.getProperty(name) : prop;
74    }

86    static {
87      InputStream in = (InputStream) AccessController.doPrivileged(
88          new PrivilegedAction()
89          {
90            public Object run() {
91              ClassLoader threadCL = Thread.currentThread().getContextClassLoader();
92              if (threadCL != null) {
93                return threadCL.getResourceAsStream(CONFIGURATION_FILE);
94              }
95              else {
96                return ClassLoader.getSystemResourceAsStream(CONFIGURATION_FILE);
97              }
98            }
99          });
100 
101     if (null != in) {
102       try {
103         configuration.load(in);
104         in.close();
105       }
106       catch (java.io.IOException e) {
107         // ignored
108       }
109     }
110 
111     showLogName = getBooleanProperty(CONFIGURATION_PREFIX + "showLogName", showLogName);
112     showLevel = getBooleanProperty(CONFIGURATION_PREFIX + "showLevel", showLevel);
113   }

115   protected int currentLogLevel = LEVEL_INFO;

117   private Class type;

119   private boolean screenOrAction = false;

121   Install4jLogger(String name) {
122     this.name = name;
123 
124     // install4j's Util.log* helpers need a type (or object) to determine name, so try and load the type here
125     this.type = loadType();
126     if (type != null) {
127       screenOrAction = Screen.class.isAssignableFrom(type) || Action.class.isAssignableFrom(type);
128     }
129 
130     // Set log level from properties
131     String level = getStringProperty(CONFIGURATION_PREFIX + "logger." + name);
132     int i = name.lastIndexOf(".");
133     while (null == level && i > -1) {
134       name = name.substring(0, i);
135       level = getStringProperty(CONFIGURATION_PREFIX + "logger." + name);
136       i = name.lastIndexOf(".");
137     }
138 
139     if (null == level) {
140       level = getStringProperty(CONFIGURATION_PREFIX + "level");
141     }
142 
143     if ("all".equalsIgnoreCase(level)) {
144       this.currentLogLevel = LEVEL_ALL;
145     }
146     else if ("trace".equalsIgnoreCase(level)) {
147       this.currentLogLevel = LEVEL_TRACE;
148     }
149     else if ("debug".equalsIgnoreCase(level)) {
150       this.currentLogLevel = LEVEL_DEBUG;
151     }
152     else if ("info".equalsIgnoreCase(level)) {
153       this.currentLogLevel = LEVEL_INFO;
154     }
155     else if ("warn".equalsIgnoreCase(level)) {
156       this.currentLogLevel = LEVEL_WARN;
157     }
158     else if ("error".equalsIgnoreCase(level)) {
159       this.currentLogLevel = LEVEL_ERROR;
160     }
161     else if ("off".equalsIgnoreCase(level)) {
162       this.currentLogLevel = LEVEL_OFF;
163     }
164   }

166   private Class loadType() {
167     try {
168       ClassLoader cl = Thread.currentThread().getContextClassLoader();
169       return cl.loadClass(name);
170     }
171     catch (Exception e) {
172       return null;
173     }
174   }

176   private void log(final int level, final String message, final Throwable t) {
177     if (!isLevelEnabled(level)) {
178       return;
179     }
180 
181     StringBuilder buff = new StringBuilder(32);
182 
183     if (showLevel) {
184       switch (level) {
185         case LEVEL_TRACE:
186           buff.append("TRACE");
187           break;
188         case LEVEL_DEBUG:
189           buff.append("DEBUG");
190           break;
191         case LEVEL_INFO:
192           buff.append("INFO");
193           break;
194         case LEVEL_WARN:
195           buff.append("WARN");
196           break;
197         case LEVEL_ERROR:
198           buff.append("ERROR");
199           break;
200       }
201       buff.append(' ');
202     }
203 
204     // Only append name if no type associated
205     if (type == null && showLogName) {
206       buff.append(name).append(" - ");
207     }
208 
209     buff.append(message);
210 
211     // If the logger if for a screen or action, null the source to use the current (which should be the screen or action)
212     Object source = type;
213     if (screenOrAction) {
214       source = null;
215     }
216 
217     // Hand over the details to install4j's Util helper
218     if (level == LEVEL_ERROR) {
219       Util.logError(source, buff.toString());
220     }
221     else {
222       Util.logInfo(source, buff.toString());
223     }
224     if (t != null) {
225       Util.log(t);
226     }
227   }

229   private void formatAndLog(int level, String format, Object arg1, Object arg2) {
230     if (!isLevelEnabled(level)) {
231       return;
232     }
233     FormattingTuple tp = MessageFormatter.format(format, arg1, arg2);
234     log(level, tp.getMessage(), tp.getThrowable());
235   }
236 
237   private void formatAndLog(int level, String format, Object[] argArray) {
238     if (!isLevelEnabled(level)) {
239       return;
240     }
241     FormattingTuple tp = MessageFormatter.arrayFormat(format, argArray);
242     log(level, tp.getMessage(), tp.getThrowable());
243   }
